  7class CalculatorBase:
  8    """
  9    Handles the core mathematical operations of the calculator.
 10
 11    This class contains methods to append input values, clear the entry,
 12    and evaluate mathematical expressions.
 13    """
 14
 15    def __init__(self) -> None:
 16        """Initialize the calculator with an empty entry value."""
 17        self.entry_value: str = ""
 18
 19    def append(self, value: Union[str, int, float]) -> None:
 20        """Append a value to the current entry."""
 21        self.entry_value += str(value)
 22
 23    def clear(self) -> None:
 24        """Clear the current entry."""
 25        self.entry_value = ""
 26
 27    def solve(self) -> Union[str, float]:
 28        """
 29        Evaluate the current mathematical expression.
 30
 31        Returns:
 32            str: 'Error' if the expression is invalid.
 33            float: The result of the valid mathematical expression.
 34        """
 35        try:
 36            return float(eval(self.entry_value))
 37        except Exception:
 38            return "Error"

117class ConversionContext:
118    """
119    Manages the current conversion strategy and executes it.
120
121    Implements the **Strategy Pattern** by dynamically assigning
122    the conversion strategy at runtime.
123    """
124
125    def __init__(self) -> None:
126        self.strategy: Optional[ConversionStrategy] = None
127
128    def set_strategy(self, strategy: ConversionStrategy) -> None:
129        """
130        Set the conversion strategy.
131
132        Args:
133            strategy (ConversionStrategy): The strategy to set.
134        """
135        self.strategy = strategy
136
137    def execute_conversion(self, value: float) -> float:
138        """
139        Execute the current strategy's conversion.
140
141        Args:
142            value (float): The value to convert.
143
144        Returns:
145            float: The converted value.
146
147        Raises:
148            ValueError: If no strategy is set.
149        """
150        if not self.strategy:
151            raise ValueError("No conversion strategy set.")
152        return self.strategy.convert(value)

201class AppMediator:
202    """
203    Central mediator for managing the application state and communication.
204
205    Implements the **Mediator Pattern** to reduce coupling and circular
206    dependency between components.
207    """
208
209    def __init__(self, master: Tk) -> None:
210        self.calculator = CalculatorBase()
211        self.conversion_context = ConversionContext()
212        self.mode: CalculatorMode = StandardMode()
213        self.display = Display(master, self)
214        self.button_manager = ButtonManager(master, self)
215
216    def set_standard_mode(self) -> None:
217        """Set the calculator to Standard mode."""
218        self.mode = StandardMode()
219        self.button_manager.update_buttons()
220
221    def set_convert_mode(self) -> None:
222        """Set the calculator to Convert mode."""
223        self.mode = ConvertMode()
224        self.button_manager.update_buttons()
225
226    def handle_clear(self) -> None:
227        """Clear the calculator display."""
228        self.calculator.clear()
229        self.display.update_display("")
230
231    def handle_equal(self) -> None:
232        """Evaluate and display the result of the current equation."""
233        result = self.calculator.solve()
234        self.display.update_display(result)
235
236    def handle_append(self, value: str) -> None:
237        """Append a value to the calculator's entry."""
238        self.calculator.append(value)
239        self.display.update_display(self.calculator.entry_value)
240
241    def handle_conversion(self, operation: str) -> None:
242        """
243        Execute a conversion operation using the appropriate strategy.
244
245        Args:
246            operation (str): The name of the conversion operation.
247        """
248        try:
249            value = float(self.calculator.entry_value)
250            strategy_map = {
251                "Mi to Km": MilesToKmStrategy(),
252                "Km to Mi": KmToMilesStrategy(),
253                "C to F": CelsiusToFahrenheitStrategy(),
254                "F to C": FahrenheitToCelsiusStrategy(),
255                "In to Cm": InchesToCentimetersStrategy(),
256                "Cm to In": CentimetersToInchesStrategy(),
257                "Min to Sec": MinutesToSecondsStrategy(),
258                "Sec to Min": SecondsToMinutesStrategy()
259            }
260            self.conversion_context.set_strategy(strategy_map[operation])
261            result = self.conversion_context.execute_conversion(value)
262            self.display.update_display(result)
263        except ValueError:
264            self.display.update_display("Error")

269class Display:
270    """Handles the display and menu."""
271
272    def __init__(self, master: Tk, mediator: AppMediator) -> None:
273        self.mediator = mediator
274        self.equation = StringVar(value="")
275        self.create_display(master)
276        self.create_menu(master)
277        self.configure_window(master)
278
279    def configure_window(self, master: Tk) -> None:
280        """Configure the main tkinter window."""
281        master.title("Calculator")
282        master.geometry('380x570+0+0')
283        master.config(bg='#484F2B')
284        master.resizable(False, False)
285
286    def create_display(self, master: Tk) -> None:
287        """Create the display entry widget."""
288        self.entry = Entry(
289            master, width=17, bg='#AEBD93', font=('Helvetica Bold', 28),
290            relief='flat', textvariable=self.equation
291        )
292        self.entry.place(x=10, y=50, width=360, height=70)
293
294    def create_menu(self, master: Tk) -> None:
295        """Create the menu for switching modes."""
296        menu = Frame(master, bg='#484F2B')
297        menu.place(x=10, y=10, width=380, height=30)
298
299        Button(
300            menu, text="Standard", bg='#7A8450', fg='white',
301            command=self.mediator.set_standard_mode, relief='flat'
302        ).pack(side='left', padx=5)
303
304        Button(
305            menu, text="Convert", bg='#7A8450', fg='white',
306            command=self.mediator.set_convert_mode, relief='flat'
307        ).place(x=275)
308
309    def update_display(self, value: Union[str, float]) -> None:
310        """Update the display with a given value."""
311        if isinstance(value, (float, int)):
312            self.equation.set(f"{value:.3f}")
313        else:
314            self.equation.set(str(value))

317class ButtonManager:
318    """Handles button creation and management."""
319
320    def __init__(self, master: Tk, mediator: AppMediator) -> None:
321        self.mediator = mediator
322        self.button_frame = Frame(master, bg='#484F2B')
323        self.button_frame.place(x=10, y=125, width=360, height=440)
324        self.update_buttons()
325
326    def update_buttons(self) -> None:
327        """Create buttons dynamically based on the current mode."""
328        for widget in self.button_frame.winfo_children():
329            widget.destroy()
330
331        buttons = self.mediator.mode.create_buttons()
332        for text, row, col in buttons:
333            if "to" in text:
334                self.create_conversion_button(text, row, col)
335            else:
336                self.create_standard_button(text, row, col)
337
338    def create_standard_button(self, text: str, row: int, col: int) -> None:
339        """Create a standard calculator button."""
340        def handle_command() -> None:
341            if text == 'C':
342                self.mediator.handle_clear()
343            elif text == '=':
344                self.mediator.handle_equal()
345            else:
346                self.mediator.handle_append(text)
347
348        Button(
349            self.button_frame, width=7, height=4, text=text, relief='flat',
350            bg='#7A8450', activebackground='#AEBD93', fg='white',
351            bd=0, highlightbackground='#484F2B', highlightcolor='#7A8450',
352            command=handle_command
353        ).grid(column=col, row=row, padx=4, pady=4)
354
355    def create_conversion_button(self, text: str, row: int, col: int) -> None:
356        """Create a conversion calculator button."""
357        def handle_conversion() -> None:
358            self.mediator.handle_conversion(text)
359
360        Button(
361            self.button_frame, width=7, height=4, text=text, relief='flat',
362            bg='white', activebackground='white', fg='black',
363            bd=0, highlightbackground='#484F2B', highlightcolor='#7A8450',
364            command=handle_conversion
365        ).grid(column=col, row=row, padx=4, pady=4)

336            else:
337                self.create_standard_button(text, row, col)
